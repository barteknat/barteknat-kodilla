16.2.1
SELECT I.SUMMARY AS TASK_LIST, IL.NAME AS TASK_NAME
FROM ISSUES I
JOIN ISSUESLISTS IL ON I.ISSUESLIST_ID = IL.ID;

16.2.2
SELECT I.SUMMARY AS TASK_LIST, U.FIRSTNAME, U.LASTNAME
FROM ISSUES I
JOIN USERS U ON I.USER_ID_ASSIGNEDTO = U.ID
ORDER BY I.ISSUESLIST_ID;

16.2.3
SELECT U.FIRSTNAME, U.LASTNAME, COUNT(*) AS NUMBER_OF_TASKS
FROM USERS U JOIN ISSUES I ON U.ID = I.USER_ID_ASSIGNEDTO
GROUP BY I.USER_ID_ASSIGNEDTO
ORDER BY NUMBER_OF_TASKS;
